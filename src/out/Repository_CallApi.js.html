import axios from "axios";

const api = axios.create({
    baseURL: `http://localhost:3000/`
});

/**
 * Retrieve user activity
 * @param {string} id
 * @returns {Array}
 */
 export const getUserActivity = async (id) => {
    try {
      const res = await api.get(`/user/${id}/activity`);
      return res.data;
    } catch (e) {
      console.log(e);
    }
  };
